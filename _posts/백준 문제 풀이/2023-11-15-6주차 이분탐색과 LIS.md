  ```cpp
  #include <bits/stdc++.h>

  int T,n,m;
  
  bool check(vector<int> &a, int i){
  	
  	int l=0;
  	int r= a.size()-1;
  	int mid;
  	while(l<=r){
  		mid=(l+r)/2;
  		if(a[mid]>i) r=mid-1;
  		else if(a[mid]==i) return 1;
  		else l=mid+1;
  	}
  	
  	return 0;
  }
  
  int main(){

  	while(T--){
  		cin >>  n;

  			cin >> a[i];

  			cin >> b[i];

  		cout << check(a, i) <<"\\n";
  		
  	}
  	
  	
  	return 0;
  ```

  ```cpp
  //보석 상자 나눠서 카운팅  
  bool check(ll mid){ 
  	ll num=0; // 그룹 크기

  		num+=a[i]/mid; // mid 값으로 나눠진 몴,  보석의 총 그룹 수  
  		if(a[i]%mid) num++;// 나머지 있으면 추가 +1 
  	}
  	return n>=num;
  }
  ```

- 11053_LIS (해결)

  ```cpp
  // 틀린 코드 -> 반례 10 20 10 40 20 25 30 50
  #include <bits/stdc++.h>
  using namespace std;
  int n,ret=0;
  
  int main() {
  	cin >> n;
  	int a[n];
  
  	for(int i=0; i<n; i++){
  		cin >> a[i];
  	}
  	
  	for(int i=0; i<n; i++){
  		int temp=a[i], cnt=1; 
  		//cout<< a[i] << "의 LIS: ";	
  		for(int j=i+1; j<n; j++){
  			if( temp < a[j]) cnt++, temp=a[j]; // 옆의 수가 크다고 바로 포함해버리면 반례 존재
  		}
  		ret=max(ret, cnt);
  
  	}
  	
  	cout << ret;
  	/*
  	for(int i : ret){
  		cout<< i << " ";
  	}
  	*/
  	
  	return 0;
  }

  #include <bits/stdc++.h>
  using namespace std;
  int n,ret=0;

  	cin >> n;
  	int a[n],cnt[n];
  
  	for(int i=0; i<n; i++){
  		cin >> a[i];

  	for(int i=0; i<n; i++){
  		int maxValue=0;

  			if( a[i] > a[j] && maxValue < cnt[j]) { // a[i]를 끝으로 하는 최장수열 길이 찾 
  				maxValue=cnt[j]; 
  			}
  		}
  		cnt[i] = maxValue+1; // cnt[i] 전의 maxvalue에 +1 
  		ret=max(ret, cnt[i]);
  
  	}
  	
  	cout << ret;
  ```
