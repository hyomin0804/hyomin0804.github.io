int n,a,depth;
vector<int> v;

int visited[1050];

void go(int depth){
	int a=0;
	while(visited[a]) a+=1;

	int start=a;
		
	for(int i=0; i<=(1<<n)-1 ; i += (1 << depth)){
		ret[depth].push_back(v[start+i]);

}

	cin >> n;

		cin >>a;

	return 0;

#include <bits/stdc++.h>
using namespace std; 
vector<int> ret[14];
int n, a[1030];
void go(int s, int e, int level){ 
    if(s > e) return; 
    if(s == e){ 
        ret[level].push_back(a[s]); 
        return;
    }
    int mid = (s + e) / 2;
    ret[level].push_back(a[mid]);
    go(s, mid - 1, level + 1); 
    go(mid + 1, e, level + 1); 
    return;
}
int main(){
    cin >> n;
    int _end = (int)pow(2, n) - 1; 
    for(int i = 0; i < _end; i++){
        cin >> a[i]; 
    }
    go(0, _end, 1); // start, end, depth
    for(int i = 1; i <= n; i++){
        for(int j : ret[i]){
            cout << j << " ";
        }
        cout << "\\n";
    }
    return 0;
}
